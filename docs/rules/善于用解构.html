# 善于用解构

  • 精简前
import React from "react";
function TestComponent(props) {
  return (
    <div>
      <p>{props.name}</p>
      <p>{props.age}</p>
    </div>
  );
}
  • 精简后
import React from "react";
function TestComponent(props) {
  const { name, age } = props;

  return (
    <div>
      <p>{name}</p>
      <p>{age}</p>
    </div>
  );
}
