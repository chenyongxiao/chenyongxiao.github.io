# 开关简化

//bad
switch (data) {
  case 1:
    test1();
    break;
  case 2:
    test2();
    break;
  case 3:
    test3();
    break;
  default:
    break;
}

//better
var data = {
  1: test1,
  2: test2,
  3: test3,
};
data[type] && data[type]();
